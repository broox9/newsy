import React from 'react';
import propTypes from 'prop-types';
import axios, {get} from 'axios';
import './styles';
 
//actions
import {fetchStoriesFromApi} from './actionCreators';
 
// Components
import SearchBox from './components/SearchBox';
import SearchResult from './components/SearchResults';
import NewsFeed from './components/NewsFeed/';
 
export default class App extends React.Component {
  constructor() {
    super();
    this.state = {
      results: [],
      error: null,
      stories: []
    };
  }
  getChildContext() {
    return { store: this.props.store }
  }
 
  componentDidMount() {
    const store = this.props.store;
    this.unsubscribe = store.subscribe(() => {
      const {search: {results, error}, news: {stories}} = store.getState();
      this.setState({results, error, stories});
    });
  }
 
  componentWillUnmount() {
    this.unsubscribe();
  }
  
  render() {
    const {error, results, stories } = this.state;
    const ErrObj = this.state.error ? (<pre>{error.code} - {error.message}</pre>) : null;
    return (
      <section>
        {ErrObj}
        <SearchBox />
        <button onClick={() => this.props.store.dispatch(fetchStoriesFromApi('cnn'))} >NEws</button>
        <SearchResult items={results} />
        <NewsFeed items={stories} />
      </section>
    )
  }
}
 
App.childContextTypes = {
  store: propTypes.object,
}
